With params

Using object

editable 
<editable-label id="with-object-params" data-name="name" data-pk="1" data-url="./backend.json">editable</editable-label>
<script>
    domready(()=>{
        document.querySelector("editable-label#with-object-params").params={param1: "param1val", param2: "param2val"};
    })
</script>

Using function

editable 
<editable-label id="with-function-params" data-name="name" data-pk="1" data-url="./backend.json">editable</editable-label>
<script>
    domready(()=>{
        document.querySelector("editable-label#with-function-params").params = function(params){
            return {
                pk: params.pk,
                name: 'new name',
                value: params.value,
                param1: 'new param val',
            }
        };
    })
</script>
